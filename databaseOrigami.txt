create schema eproject_aptech;
use eproject_aptech;

CREATE TABLE users (
  user_id INTEGER PRIMARY KEY AUTO_INCREMENT,
  username VARCHAR(255),
  password VARCHAR(255),
  email VARCHAR(255)
);
CREATE TABLE origami_diagrams (
  model_id INTEGER PRIMARY KEY,
  origami_name VARCHAR(255),
  url_img VARCHAR(255),
  description TEXT,
  category VARCHAR(255)
);

CREATE TABLE origami_guide (
  model_id INTEGER,
  step_number INTEGER,
  img_url VARCHAR(255),
  description_guide TEXT,
  FOREIGN KEY (model_id) REFERENCES origami_diagrams(model_id)
);

CREATE TABLE feedback (
  contact_id INTEGER PRIMARY KEY,
  user_id INTEGER,
  email VARCHAR(255),
  username VARCHAR(255),
  content TEXT,
  date_create TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(user_id)
);
